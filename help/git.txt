////////СВЯЗАТЬ С ПАПКОЙ/////
git init
git remote add origin

////////////  ВЕТКИ //////////////
git branch - показать все существующие ветки 
git branch name - создать ветку
git checkout name - перейти в ветку name 
git checkout -b name - создать ветку name и перейти в нее

///////////// MERGE /////////////
git merge name - из ветки в которой находимся сливаем все в name 


git add .   - добавить все измененния
git status - глянуть какие изменения ожидают
git commit -m "comments" - залить изменения с комментарием 

заходим на ветку с которой хотим слить изменения
git merge name - нейм - это имя ветки с которой ссливаем
